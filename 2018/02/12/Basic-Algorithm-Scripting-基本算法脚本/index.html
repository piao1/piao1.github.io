function StringResverse(str){
var arr = str.spilt(“”); // 注意这里的空字符
var arr1 = arr.reverse();
return arr1.join(“”); // 注意这里的空字符
}

#计算一个数的阶乘
function factorialize(num) {
val = 1;
while (num > 1){
val = val * num;
num –;
}
return val;
}

#找出最长单词
function findLongestWord(str) {
var arr = str.split(“ “);
maxValue = arr[0];
for ( var i = 0;i < arr.length ;i++){
if (arr[i].length > maxValue.length){
maxValue = arr[i];
}
}
return maxValue.length;
}
findLongestWord(“What is the average airspeed velocity of an unladen swallow”);

#句中单词首字母大写
function titleCase(str){
var arr = str.toLowerCase().split(“ “);
for (var i = 0; i < arr.length; i++){
var char = arr[i].charAt(0);
arr[i] = arr[i].replace(char,function(s){return s.toUpperCase();});
}
str = arr.join(“ “);
return str;
}
titleCase(“I’m a little tea pot”);

#找出多个数组中的最大数
function largestOfFour(arr) {
var arr1 = new Array;
for(var i = 0;i < arr.length;i++){
arr1[i] = arr[i][0];
for (var j = 0;j < arr[i].length; j++){
if(arr[i][j] > arr1[i]){
arr1[i] = arr[i][j];
}
}
}
return arr1;
}
largestOfFour([[4, 5, 1, 3], [13, 27, 18, 26], [32, 35, 37, 39], [1000, 1001, 857, 1]]);
